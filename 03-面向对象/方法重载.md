class Person {  
    private String name;  
    private String surname;  
  
    public String getName() {  
        if( surname != null && surname.isEmpty()){  
            return name + " " + surname;  
        } else {  
            return name;  
        }  
  
    }  
  
    public void setName(String name) {  
        this.name = name;  
    }  
  
    public  void setName(String name, String surname){  
        this.name = name;  
        this.surname = surname;  
    }  
}
